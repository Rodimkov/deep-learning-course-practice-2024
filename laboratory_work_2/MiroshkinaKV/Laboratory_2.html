class ResidualBlock(nn.Module):
    def __init__(self, in_channels, out_channels, stride=1):
        super(ResidualBlock, self).__init__()
        # Первый сверточный слой
        self.conv1 = nn.Conv2d(in_channels, out_channels, kernel_size=3, stride=stride, padding=1)
        self.bn1 = nn.BatchNorm2d(out_channels)  # Нормализация данных 
        # Второй сверточный слой
        self.conv2 = nn.Conv2d(out_channels, out_channels, kernel_size=3, stride=1, padding=1)
        self.bn2 = nn.BatchNorm2d(out_channels)  # Нормализация данных
        # Shortcut
        self.shortcut = nn.Sequential()
        if stride != 1 or in_channels != out_channels:  # Если размеры не совпадают, создаем shortcut
            self.shortcut = nn.Sequential(
                nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=stride),  # 1x1 свертка
                nn.BatchNorm2d(out_channels)
            )

    def forward(self, x):
        # ReLu и нормализация
        out = nn.ReLU()(self.bn1(self.conv1(x)))
        out = self.bn2(self.conv2(out))
        # Добавление shortcut
        out += self.shortcut(x)
        # Применяем ReLU к выходу
        out = nn.ReLU()(out)
        return out

class ResNet(nn.Module):
    def __init__(self, block, num_blocks, num_classes=10):
        super(ResNet, self).__init__()
        self.in_channels = 64  # Начальное количество каналов
        # Входной сверточный слой
        self.conv1 = nn.Conv2d(3, 64, kernel_size=3, stride=1, padding=1)
        self.bn1 = nn.BatchNorm2d(64)
        # Слои сети
        self.layer1 = self._make_layer(block, 64, num_blocks[0], stride=1)
        self.layer2 = self._make_layer(block, 128, num_blocks[1], stride=2)
        self.layer3 = self._make_layer(block, 256, num_blocks[2], stride=2)
        self.layer4 = self._make_layer(block, 512, num_blocks[3], stride=2)
        # Полносвязный выходной слой
        self.fc = nn.Linear(512, num_classes)

    def _make_layer(self, block, out_channels, num_blocks, stride):
        # Шаги свертки для кадого блока
        strides = [stride] + [1] * (num_blocks - 1)  # Первому блоку задается stride, остальные stride=1
        layers = []
        for stride in strides:
            layers.append(block(self.in_channels, out_channels, stride))
            self.in_channels = out_channels  # Обновление количества каналов
        return nn.Sequential(*layers)

    def forward(self, x):
        # Проход через все слои
        out = nn.ReLU()(self.bn1(self.conv1(x)))
        out = self.layer1(out)
        out = self.layer2(out)
        out = self.layer3(out)
        out = self.layer4(out)
        out = nn.AdaptiveAvgPool2d((1, 1))(out)  # Сжимание выхода в 1x1
        out = out.view(out.size(0), -1)  # Выравнивание выхода
        out = self.fc(out)  # Последний линейный слой
        return out

def train_model(model, train_loader, criterion, optimizer, scheduler, num_epochs):
    train_losses = []
    train_accuracies = []
    for epoch in range(num_epochs):
        model.train()  # Перевод модели в режим обучения
        running_loss = 0.0
        correct = 0
        total = 0
        for images, labels in train_loader:
            images, labels = images.to(device), labels.to(device)  # Перенос данных на устройство
            optimizer.zero_grad()  # Обнуление градиентов
            outputs = model(images)  # Прямой проход
            loss = criterion(outputs, labels)  # Вычисление ошибки
            loss.backward()  # Обратное распространение
            optimizer.step()  # Шаг оптимизации
            running_loss += loss.item()
            _, predicted = torch.max(outputs.data, 1)  # Получение предсказаний
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        scheduler.step()  # Обновление скорости обучения
        epoch_loss = running_loss / len(train_loader)
        epoch_accuracy = 100 * correct / total
        train_losses.append(epoch_loss)
        train_accuracies.append(epoch_accuracy)
        print(f'Epoch [{epoch+1}/{num_epochs}], Loss: {epoch_loss:.4f}, Accuracy: {epoch_accuracy:.2f}%')
    return train_losses, train_accuracies

def evaluate_model(model, test_loader):
    model.eval()  # Перевод модели в режим оценки
    correct = 0
    total = 0
    with torch.no_grad():  # Отключение градиентов
        for images, labels in test_loader:
            images, labels = images.to(device), labels.to(device)
            outputs = model(images)
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
    accuracy = 100 * correct / total
    print(f'Test Accuracy: {accuracy:.2f}%')
    return accuracy
