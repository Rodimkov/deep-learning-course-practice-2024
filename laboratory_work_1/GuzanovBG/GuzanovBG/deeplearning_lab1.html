def result_of_visualisation(model, test_data):
    (cols, rows) = (10, 4)
    _, axes = plot.subplots(nrows=rows, ncols=cols, figsize=(21, 10))
    for col in range(cols):
        for row in range(rows):
            idx = random.randint(0, len(test_data[0]))
            image = test_data[0][idx].reshape(28, 28)
            label = test_data[1][idx]

                f" {np.argmax(prediction, axis=1)[0]}")

    def softmax(x):
        exp_x = np.exp(x - np.max(x, axis=1, keepdims=True))
        return exp_x / np.sum(exp_x, axis=1, keepdims=True)

    
    def relu(x):
        return np.maximum(0, x)

    def relu_derivative(x):
        return np.where(x > 0, 1, 0)

    def layer(X, W, B):
        return np.dot(X, W) + B

class Network:

    def __init__(self, input_size, hidden_size, output_size, learning_rate):
        self.output_size = output_size
        self.learning_rate = learning_rate
        self.input_size = input_size
        self.hidden_size = hidden_size
        
        #   (Uniform Xavier Initialization)
        self.b1 = np.zeros((1, hidden_size))
        self.b2 = np.zeros((1, output_size))
        s1 = np.sqrt(6 / (input_size + hidden_size))
        s2 = np.sqrt(6 / (hidden_size + output_size))
        self.W1 = np.random.uniform(-s1, s1, [input_size, hidden_size])
        self.W2 = np.random.uniform(-s2, s2, [hidden_size, output_size])

    def forward(self, x):
        self.z1 = layer(x, self.W1, self.b1)
        self.a1 = relu(self.z1)
        self.z2 = layer(self.a1, self.W2, self.b2)
        y_pred = softmax(self.z2)
        return y_pred
    
    def backward(self, X, y):
        y_pred = self.forward(X)
        dz2 = (y_pred - y) / X.shape[0]
        dW2 = np.dot(self.a1.T, dz2)
        db2 = np.sum(dz2, axis=0, keepdims=True)

        dz1 = relu_derivative(self.z1) * np.dot(dz2, self.W2.T)
        dW1 = np.dot(X.T, dz1)
        db1 = np.sum(dz1, axis=0, keepdims=True)

        self.W1 -= self.learning_rate * dW1
        self.b1 -= self.learning_rate * db1
        self.W2 -= self.learning_rate * dW2 
        self.b2 -= self.learning_rate * db2

for epoch in (range(epochs)):
    indexes = np.random.permutation(len(train_images))
    images = train_images[indexes]
    labels = train_labels[indexes]
    time_start = time.time()

    for i in range(0, len(images), batch_size):
        images_batch = images[i:i + batch_size]
        labels_batch = labels[i:i + batch_size]
        model.backward(images_batch, labels_batch)

    y_pred = model.forward(images)
    accuracy = np.mean(np.argmax(y_pred, axis=1) == np.argmax(labels, axis=1))
    crossentropy = np.mean(-np.sum(labels * np.log(y_pred), axis = 1))
    run_time = time.time() - time_start
    print(f": {epoch + 1};  : {crossentropy}; : {accuracy:.4f}; : {run_time:.2f}")
